def get_user(x):
    t = x[(x.find("dir/")+4):]
    return t[:t.find("/")]

def get_mail_folder(x):
    t = x[(x.find("dir/")+4):]
    t2 = t[(t.find("/")+1):]
    return t2[:t2.find("/")]

def get_date(x):
    return pd.to_datetime(x.split("Date:")[1])

def get_dayweek(x):
    return x.dayofweek

def get_year(x):
    return x.year

def get_hour(x):
    return x.hour

def get_month(x):
    return x.month
